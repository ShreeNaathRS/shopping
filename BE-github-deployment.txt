springboot-app.service file will allow us to start, stop, and restart it cleanly using systemctl.
sudo nano /etc/systemd/system/springboot-app.service
and paste
[Unit]
Description=Spring Boot Backend Service
After=network.target

[Service]
User=ubuntu
WorkingDirectory=/home/ubuntu/products
ExecStart=/usr/bin/java -jar /home/ubuntu/products/target/products-0.0.1-SNAPSHOT.jar
SuccessExitStatus=143
Restart=always
RestartSec=5

[Install]
WantedBy=multi-user.target


sudo systemctl daemon-reload
sudo systemctl enable springboot-app
sudo systemctl status springboot-app
